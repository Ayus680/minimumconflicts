import random
import matplotlib.pyplot as plt
import networkx as nx
import itertools

class Map:
    def __init__(self):
        self.domains = []
        self.regions = []

    def draw_map(self, node_size=800, font_size=12):
        node_colors = [region.color for region in self.regions]

        G = nx.Graph()
        positions = {}

        for region in self.regions:
            G.add_node(region.name)
            positions[region.name] = (region.posX, region.posY)

        for region in self.regions:
            for neighbor in region.neighbors:
                G.add_edge(region.name, neighbor.name)

        plt.figure(figsize=(10, 8))
        nx.draw(
            G,
            pos=positions,
            with_labels=True,
            node_size=node_size,
            font_size=font_size,
            node_color=node_colors,
        )
        plt.title("Map of India (Regions and Neighbors)")
        plt.show()

class Region:
    def __init__(self, name, posX, posY):
        self.name = name
        self.posX = posX
        self.posY = posY
        self.neighbors = []
        self.color = None

def min_conflicts(india_map, max_steps):
    # Assign random colors initially
    for region in india_map.regions:
        region.color = random.choice(india_map.domains)

    for step in range(max_steps):
        # Identify conflicted regions
        conflicted_regions = [
            region for region in india_map.regions if any(
                neighbor.color == region.color for neighbor in region.neighbors
            )
        ]

        if not conflicted_regions:
            return True, step
 region = random.choice(conflicted_regions)
        conflict_counts = {}
        for color in india_map.domains:
            conflict_counts[color] = sum(
                1 for neighbor in region.neighbors if neighbor.color == color
            )

        region.color = min(conflict_counts, key=conflict_counts.get)
    return False, max_steps
